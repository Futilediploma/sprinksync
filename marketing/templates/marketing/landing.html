# marketing/views.py import os import requests from django.conf import settings from django.shortcuts import render from django.core.mail import send_mail from .forms import InterestForm def landing_page(request): form = InterestForm(request.POST or None) success = False if request.method == 'POST' and form.is_valid(): interest = form.save() # transactional emails… send_mail(...) # add to Brevo contacts api_url = 'https://api.brevo.com/v3/contacts' headers = { 'Accept': 'application/json', 'Content-Type': 'application/json', 'api-key': os.environ.get('BREVO_API_KEY', ''), } payload = { 'email': interest.email, 'attributes': {'FNAME': interest.name or ''}, 'listIds': [int(os.environ.get('BREVO_LIST_ID', 0))], 'updateEnabled': True, } resp = requests.post(api_url, headers=headers, json=payload) # Debug output print("=== Brevo API call ===") print("URL:", api_url) print("List ID:", os.environ.get('BREVO_LIST_ID')) print("Status:", resp.status_code) print("Response body:", resp.text) success = True return render(request, 'marketing/landing.html', {'form': form, 'success': success})
